git init = za kreiranje direktorija
git status
git add . = za dodavanje svih filova za commit
git rm --cached "File name" = oklanjanje iz commita
git commit -m "Opis" = commitaj u lokalni repozitorij
git checkout "File name" = anuliranje promjena (jednog file ili vise datoteka zavisi jeli modificirano jedan file ili napravljena nova datoteka)
