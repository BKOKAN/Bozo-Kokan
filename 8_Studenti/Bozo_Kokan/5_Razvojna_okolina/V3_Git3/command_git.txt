git init = za kreiranje direktorija
git status
git add . = za dodavanje svih filova za commit
git rm --cached "File name" = oklanjanje iz commita
git commit -m "Opis" = commitaj u lokalni repozitorij
git log = provjera svih commitova
git branch "branch name" = kreiranje novog brancha
git checkout "File name" = anuliranje promjena (jednog file ili vise datoteka zavisi jeli modificirano jedan file ili napravljena nova datoteka)
git merge "ime brancha" = pozicionirat se na master/main sa commandong checkout nakon toga commanda merge za spajanje 2 bramcha
git branch -d "ime brancha" = brisanje brancha ali prije toga se pozicionirat na master/main branch
git remote add origin "https link od github repozitorija" = dodavanje github repozitorija
